CREATE DATABASE poe_user_data;
GRANT ALL 

CREATE TABLE char_types(
id PRIMARY KEY NOT NULL,
name CHAR(25) NOT NULL
)

CREATE TABLE accounts(
id SERIAL NOT NULL,
ggg_account_name CHAR VARYING(50) NOT NULL,
gg_sessid CHAR(32)
) 


CREATE TABLE characters(
id SERIAL PRIMARY KEY, 
name CHAR VARYING(50) NOT NULL,
account_id INT NOT NULL,
level INT NOT NULL,
class CHAR(25) NOT NULL,
league CHAR(25) NOT NULL,
classId INT NOT NULL,
ascendancyClass INT NOT NULL
);
